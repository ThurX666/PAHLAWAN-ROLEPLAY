public OnVipExpired(playerid)
{
    new ucpname[MAX_PLAYER_NAME];
    
    // Asumsikan helper get name
    format(ucpname, sizeof(ucpname), "PlayerUCP"); // Ganti dg fungsi GetUserUCP(playerid)
    
    // Get VIP Tier dari enum/variable player (Bronze, Silver, Gold)
    new vipLevel = 1; // contoh 1 = Bronze
    new vipTierStr[32];
    
    switch(vipLevel)
    {
        case 1: strpack(vipTierStr, "Bronze");
        case 2: strpack(vipTierStr, "Silver");
        case 3: strpack(vipTierStr, "Gold");
        default: strpack(vipTierStr, "Bronze");
    }
    
    new query[1024];
    mysql_format(handle, query, sizeof(query), "INSERT INTO ucp_inbox_messages (username, title, message, type, is_read, template, metadata) VALUES ('%e', 'Pemberitahuan: Status VIP %e Berakhir', '', 'System', 0, 'VipExpired', '{\"vipTier\": \"%e\"}')",
        ucpname, vipTierStr, vipTierStr);
        
    mysql_tquery(handle, query, "", "");
    
    SendClientMessage(playerid, -1, "{FFFF00}[INFO]: {FFFFFF}Status VIP anda telah berakhir. Cek UCP Inbox untuk info lebih lanjut.");
    return 1;
}
