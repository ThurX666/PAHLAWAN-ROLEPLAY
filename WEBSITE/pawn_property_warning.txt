public OnPropertyInactivityWarning(const ucpname[], const propertyType[], propertyId, const propertyZone[], inactiveDays, daysRemaining)
{
    new query[1024];
    new titleStr[128];
    format(titleStr, sizeof(titleStr), "Peringatan Sistem: Inaktivitas Properti (%s)", propertyType);
    
    // Escape string values if needed (propertyType, propertyZone) but using mysql_format %e is safe enough.
    mysql_format(handle, query, sizeof(query), 
        "INSERT INTO ucp_inbox_messages (username, title, message, type, is_read, template, metadata) "
        "VALUES ('%e', '%e', '', 'System', 0, 'PropertyInactivityWarning', "
        "'{\"propertyType\": \"%e\", \"propertyId\": %d, \"propertyZone\": \"%e\", \"inactiveDays\": %d, \"daysRemaining\": %d}')",
        ucpname, titleStr, propertyType, propertyId, propertyZone, inactiveDays, daysRemaining);
        
    mysql_tquery(handle, query, "", "");
    return 1;
}
