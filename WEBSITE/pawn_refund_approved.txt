new MySQL:handle; // Asumsikan koneksi db

forward OnRefundApproved(const ucpname[], const assetName[], const commandToUse[], const locationToClaim[]);

public OnRefundApproved(const ucpname[], const assetName[], const commandToUse[], const locationToClaim[])
{
    new query[1024];
    
    // Escape string values using mysql_format %e
    mysql_format(handle, query, sizeof(query), 
        "INSERT INTO ucp_inbox_messages (username, title, message, type, is_read, template, metadata) "
        "VALUES ('%e', 'Informasi: Request Refund Disetujui', '', 'System', 0, 'RefundApproved', "
        "'{\"assetName\": \"%e\", \"command\": \"%e\", \"locationToClaim\": \"%e\"}')",
        ucpname, assetName, commandToUse, locationToClaim);
        
    mysql_tquery(handle, query, "", "");
    return 1;
}

// Fungsi ini biasanya dipanggil setelah admin secara In-Game/UCP menyetujui request Refund
CMD:acceptrefund(playerid, params[])
{
    // Cek admin auth
    
    new ucpname[32], assetName[64], commandToUse[32], locationToClaim[64];
    
    // Contoh sederhana ambil argumen
    if(sscanf(params, "s[32]s[64]s[32]s[64]", ucpname, assetName, commandToUse, locationToClaim)) 
        return SendClientMessage(playerid, -1, "USAGE: /acceptrefund [ucpname] [assetName] [commandToUse] [locationToClaim]");
    
    // Logic untuk menambahkan item ke table refund ...
    
    // Kirim notifikasi ke pelapor
    OnRefundApproved(ucpname, assetName, commandToUse, locationToClaim);
    
    SendClientMessage(playerid, -1, "{00FF00}[SUCCESS]: {FFFFFF}Refund berhasil disetujui, notifikasi Inbox telah dikirimkan ke pemain.");
    return 1;
}
