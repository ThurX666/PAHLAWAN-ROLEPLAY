#include <a_samp>
#include <a_mysql>

extern MySQL:g_SQL;

// Get total connected players
stock GetOnlinePlayersCount()
{
    new count = 0;
    // Assuming you are using y_iterate/foreach
    foreach(new i : Player) 
    {
        if(IsPlayerConnected(i)) count++;
    }
    return count;
}

// Call this function in a 1-Minute Timer (e.g., inside OnGameModeInit using SetTimer)
forward UpdateServerActivityChart();
public UpdateServerActivityChart()
{
    new query[256], players = GetOnlinePlayersCount();
    
    // We insert a new row periodically. 
    // The PHP Backend uses MAX() grouped by hour, so multiple rows are perfectly handled.
    mysql_format(g_SQL, query, sizeof(query),
        "INSERT INTO `ucp_server_activity` (`players_online`, `logged_at`) VALUES (%d, NOW())",
        players
    );
    
    mysql_tquery(g_SQL, query, "", "");
    return 1;
}
