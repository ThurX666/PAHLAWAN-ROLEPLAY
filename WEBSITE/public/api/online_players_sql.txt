-- Tabel untuk menampung data pemain yang sedang online secara langsung dari Gamemode
-- Disarankan gamemode melakukan update tabel ini setiap kali pemain login/logout dan mengupdate score/ping secara berkala (misal tiap menit)

CREATE TABLE IF NOT EXISTS `ucp_online_players` (
  `player_id` int(11) NOT NULL,
  `character_name` varchar(32) NOT NULL,
  `color` varchar(10) NOT NULL DEFAULT '#FFFFFF',
  `score` int(11) NOT NULL DEFAULT 0,
  `ping` int(11) NOT NULL DEFAULT 0,
  `login_time` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`player_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
