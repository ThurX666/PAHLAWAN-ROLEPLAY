new MySQL:handle; // Asumsikan koneksi db

forward OnPlayerReportResponded(const ucpname[], const reportedPlayer[], const violation[], const adminAction[]);

public OnPlayerReportResponded(const ucpname[], const reportedPlayer[], const violation[], const adminAction[])
{
    new query[1024];
    
    // Escape string values using mysql_format %e
    mysql_format(handle, query, sizeof(query), 
        "INSERT INTO ucp_inbox_messages (username, title, message, type, is_read, template, metadata) "
        "VALUES ('%e', 'Laporan Pemain Ditanggapi', '', 'System', 0, 'PlayerReportResponded', "
        "'{\"reportedPlayer\": \"%e\", \"violation\": \"%e\", \"adminAction\": \"%e\"}')",
        ucpname, reportedPlayer, violation, adminAction);
        
    mysql_tquery(handle, query, "", "");
    return 1;
}

// Fungsi ini biasanya dipanggil setelah admin secara In-Game/UCP (Discord/Web) merespond suatu report
// Contoh cmd in-game untuk close report:
CMD:closereport(playerid, params[])
{
    // Cek admin auth
    
    new ucpname[32], reportedPlayer[MAX_PLAYER_NAME], violation[64], adminAction[128];
    
    // Contoh sederhana ambil argumen
    if(sscanf(params, "s[32]s[24]s[64]s[128]", ucpname, reportedPlayer, violation, adminAction)) 
        return SendClientMessage(playerid, -1, "USAGE: /closereport [pelaporUCP] [terlaporName] [pelanggaran] [sanksi]");
    
    // Logic untuk update status forum/ingame report...
    
    // Kirim notifikasi ke pelapor
    OnPlayerReportResponded(ucpname, reportedPlayer, violation, adminAction);
    
    SendClientMessage(playerid, -1, "{00FF00}[SUCCESS]: {FFFFFF}Laporan berhasil ditutup, notifikasi Inbox telah dikirimkan ke pelapor.");
    return 1;
}
